    47→function extractPlaylistId(urlOrId) {
    48→  const s = String(urlOrId || '').trim();
    49→  if (!s) return null;
    50→  try {
    51→    if (s.includes('http://') || s.includes('https://')) {
    52→      const urlObj = new URL(s);
    53→      const listId = urlObj.searchParams.get('list');
    54→      if (listId) return listId;
    55→    }
    56→  } catch {}
    57→  return s;
    58→}

   108→  const app = express();

   110→  // Detrás de Cloudflare/ngrok hay 1 hop de proxy. Limitar a 1 (en vez de
   111→  // `true`) evita que un cliente malicioso pueda inyectar varias IPs en

   116→  // ── Cabeceras de seguridad ────────────────────────────────────
   117→  // Se aplican a todas las respuestas. La CSP es deliberadamente permisiva
   118→  // en img/media/font (la app carga portadas de YouTube Music y fuentes
   119→  // inline de Tailwind) pero estricta en script-src ('self' solo — sin
   120→  // 'unsafe-inline' ni 'unsafe-eval', ya que el build de Vite produce
   121→  // assets con hash y el SW está en el mismo origen).
   122→  // El HSTS solo se activa cuando la petición llega por HTTPS (en dev
   123→  // detrás de localhost no se envía para no romper el ciclo de dev).
   124→  app.use((req, res, next) => {
   125→    res.setHeader('X-Content-Type-Options', 'nosniff');
   126→    res.setHeader('X-Frame-Options', 'SAMEORIGIN');
   127→    res.setHeader('Referrer-Policy', 'no-referrer');
   128→    res.setHeader('Permissions-Policy', 'browsing-topics=(), camera=(), microphone=(), geolocation=()');
   129→    // CSP — solo para respuestas HTML/JSON (no para audio, que no la necesita
   130→    // y cuyo añadirla añadiría overhead en cada Range).
   131→    const ct = String(req.headers.accept || '');
   132→    if (ct.includes('text/html') || ct.includes('application/json') || req.path.startsWith('/api/')) {
   133→      res.setHeader(
   134→        'Content-Security-Policy',
   135→        [
   136→          "default-src 'self'",
   137→          "script-src 'self'",
   138→          "style-src 'self' 'unsafe-inline'", // Tailwind/Vite inyectan estilos inline
   139→          "img-src 'self' data: blob: https:",
   140→          "media-src 'self' https: blob: data:",
   141→          "font-src 'self' data:",
   142→          "connect-src 'self' https://oauth2.googleapis.com https://lrclib.net https://api.lyrics.ovh",
   143→          "frame-ancestors 'self'",
   144→          "base-uri 'self'",
   145→          "form-action 'self' https://accounts.google.com",
   146→        ].join('; '),
   147→      );
   148→    }
   149→    // HSTS solo en HTTPS. En Cloudflare Tunnel llega como HTTPS al edge.
   150→    if (req.secure || req.headers['x-forwarded-proto'] === 'https') {
   151→      res.setHeader('Strict-Transport-Security', 'max-age=31536000; includeSubDomains; preload');
   152→    }
   153→    next();
   154→  });
   155→  // Compresión gzip para respuestas de texto (JSON/HTML/JS/CSS). NO comprime el

   158→    filter: (req, res) => {

   163→  // ── CORS ──────────────────────────────────────────────────────
   164→  // En producción, si ALLOWED_ORIGIN no está configurado, se rechazan las
   165→  // peticiones cross-origin (fail-closed). Antes el default era '*' lo cual
   166→  // permitía que cualquier sitio web llamara a la API con credenciales.
   167→  // En desarrollo (NODE_ENV !== 'production') se usa un allowlist explícito
   168→  // de orígenes locales (Vite dev server + variantes) en vez de '*'.
   169→  const DEV_ORIGINS = [
   170→    'http://localhost:5173',     // Vite dev server (frontend)
   171→    'http://localhost:3000',     // Backend sirviendo el build
   172→    'http://127.0.0.1:5173',
   173→    'http://127.0.0.1:3000',
   174→  ];
   175→  const corsOrigin = (() => {
   176→    if (process.env.ALLOWED_ORIGIN) {
   177→      // Soporta un solo origen o una lista separada por comas.
   178→      const list = process.env.ALLOWED_ORIGIN.split(',').map((s) => s.trim()).filter(Boolean);
   179→      return list.length === 1 ? list[0] : list;

   183→      return false; // Rechazar todas las peticiones cross-origin en prod
   184→    }
   185→    return DEV_ORIGINS; // Dev: allowlist explícito (no '*')

   277→  // Caché de búsqueda en memoria (resultados, no audio). TTL 5 minutos, máx 200 entradas.

   284→      const oldest = searchCache.keys().next().value;
   285→      searchCache.delete(oldest);
   286→    }

   293→    const cacheKey = `${qRaw}|${req.query.limit ?? ''}`;
   294→    const cached = searchCache.get(cacheKey);

   323→  // ---- Letras (YouTube Music nativo + lrclib filtrado + lyrics.ovh) ----
   324→  // lrclib search devolvía el primer hit y a menudo la letra de OTRA canción.
   325→  // Ahora se puntúa por artist/title/duration y se rechaza si no cuadra.
   326→  const lyricsCache = new Map();
   327→  const LYRICS_CACHE_MAX = 500;
   328→  const lyricsCacheSet = (key, val) => {
   329→    if (lyricsCache.size >= LYRICS_CACHE_MAX) {
   330→      const oldest = lyricsCache.keys().next().value;
   331→      lyricsCache.delete(oldest);
   332→    }
   333→    lyricsCache.set(key, val);
   334→  };
   335→  const withTimeout = (p, ms) => Promise.race([p, new Promise((_, rej) => setTimeout(() => rej(new Error('timeout')), ms))]);
   336→  app.get('/api/lyrics', async (req, res) => {
   337→    const artist = String(req.query.artist || '').trim();
   338→    const title = String(req.query.title || '').trim();
   339→    const album = String(req.query.album || '').trim();
   340→    const duration = String(req.query.duration || '').trim();
   341→    const id = String(req.query.id || '').trim();
   342→    const syncOnly = String(req.query.sync || '') === '1';
   343→    if (!artist || !title) return res.status(400).json({ error: 'Se requieren artist y title.' });
   344→
   345→    const cleanTitle = cleanLyricQuery(title);
   346→    const key = `${id || ''}|${artist}|${title}`.toLowerCase();
   347→    const cached = lyricsCache.get(key);
   348→    if (cached && Date.now() - cached.at < 7 * 86400000) {
   349→      if (!syncOnly || cached.data.synced) return res.json(cached.data);
   350→    }
   351→
   352→    const head = { 'User-Agent': 'VelocityMusic/1.0 (personal use)' };
   353→    const queryMeta = { artist, title: cleanTitle || title, duration };
   354→    const finish = (data) => { lyricsCacheSet(key, { data, at: Date.now() }); return res.json(data); };
   355→
   356→    const packFromCandidate = (c) => {
   357→      if (!c) return null;
   358→      const synced = c.syncedLyrics || c.synced || null;
   359→      const plain = c.plainLyrics || c.plain || (synced ? plainFromSynced(synced) : null);
   360→      if (!synced && !plain) return null;
   361→      return { source: 'lrclib', synced: synced || null, plain: plain || null };
   362→    };
   363→
   364→    // GET exacto de lrclib (artist+title+duration) — más fiable que search.
   365→    const lrcGet = (ms) => (async () => {
   366→      try {
   367→        const u = new URL('https://lrclib.net/api/get');
   368→        u.searchParams.set('artist_name', artist);
   369→        u.searchParams.set('track_name', cleanTitle || title);
   370→        if (album) u.searchParams.set('album_name', album);
   371→        if (duration) u.searchParams.set('duration', String(Math.round(Number(duration) || duration)));
   372→        const r = await withTimeout(fetch(u, { headers: head }), ms);
   373→        if (!r.ok) return null;
   374→        const d = await r.json();
   375→        if (!d || !(d.syncedLyrics || d.plainLyrics)) return null;
   376→        // Aun el GET puede devolver basura: puntuar.
   377→        const scored = pickBestLyricsCandidate(queryMeta, [d], 45);
   378→        return scored ? packFromCandidate(scored.candidate) : null;
   379→      } catch { return null; }
   380→    })();
   381→
   382→    // SEARCH de lrclib: ranking estricto, NUNCA arr[0] a ciegas.
   383→    const lrcSearch = (ms) => (async () => {
   384→      try {
   385→        const u = new URL('https://lrclib.net/api/search');
   386→        u.searchParams.set('q', `${cleanTitle || title} ${artist}`);
   387→        const r = await withTimeout(fetch(u, { headers: head }), ms);
   388→        if (!r.ok) return null;
   389→        const arr = await r.json();
   390→        if (!Array.isArray(arr) || !arr.length) return null;
   391→        const picked = pickBestLyricsCandidate(queryMeta, arr, 55);
   392→        return picked ? packFromCandidate(picked.candidate) : null;
   393→      } catch { return null; }
   394→    })();
   395→
   396→    const bestLrc = async (ms) => {
   397→      const [g, se] = await Promise.all([lrcGet(ms), lrcSearch(ms)]);
   398→      // Preferir el que tenga synced y, a igualdad, el GET.
   399→      if (g?.synced) return g;
   400→      if (se?.synced) return se;
   401→      return g || se || null;
   402→    };
   403→
   404→    // ── Modo sync: solo lrclib bien emparejado ──
   405→    if (syncOnly) {
   406→      const lrc = await bestLrc(15000);
   407→      if (lrc?.synced) return finish(lrc);
   408→      if (lrc?.plain) return finish({ ...lrc, synced: null });
   409→      return res.status(404).json({ error: 'Sin letra sincronizada.' });
   410→    }
   411→
   412→    // ── Modo rápido: lrclib corto + YT nativo + ovh ──
   413→    const ytP = (id && typeof lyricsByIdImpl === 'function')
   414→      ? withTimeout(lyricsByIdImpl(id), 4000).then((p) => (p && p.trim() ? p.trim() : null)).catch(() => null)
   415→      : Promise.resolve(null);
   416→
   417→    const lrc = await bestLrc(4500);
   418→    if (lrc?.synced) return finish(lrc);
   419→
   420→    const yt = await ytP;
   421→    if (yt) {
   422→      // Si lrclib trae plain que no se parece al de YT, ignorarlo.
   423→      if (lrc?.plain && lyricsOverlapRatio(yt, lrc.plain) < 0.35) {
   424→        return finish({ source: 'youtube-music', synced: null, plain: yt });
   425→      }
   426→      return finish({ source: 'youtube-music', synced: null, plain: yt });
   427→    }
   428→    if (lrc?.plain) return finish(lrc);
   429→
   430→    try {
   431→      const r = await withTimeout(
   432→        fetch(`https://api.lyrics.ovh/v1/${encodeURIComponent(artist)}/${encodeURIComponent(cleanTitle || title)}`, { headers: head }),
   433→        3500
   434→      );
   435→      if (r.ok) {
   436→        const d = await r.json();
   437→        if (d?.lyrics?.trim()) return finish({ source: 'lyrics.ovh', synced: null, plain: d.lyrics.trim() });
   438→      }
   439→    } catch {}
   440→    return res.status(404).json({ error: 'Letra no encontrada.' });
   441→  });
   442→
   443→  // ---- Búsqueda combinada (canciones + álbumes + artistas) ----

   469→    if (detailCache.size >= DETAIL_CACHE_MAX) {
   470→      const oldest = detailCache.keys().next().value;
   471→      detailCache.delete(oldest);

  1206→/** Envuelve un handler async mapeando errores tipados a su código HTTP. */
  1207→function wrap(handler, ErrorType) {
  1208→  return async (req, res) => {
  1209→    try {
  1210→      await handler(req, res);
  1211→    } catch (err) {
  1212→      if (ErrorType && err instanceof ErrorType) {
  1213→        return res.status(err.status).json({ error: err.message });
  1214→      }
  1215→      return res.status(500).json({ error: 'Error interno.' });
  1216→    }
  1217→  };
